一般的fuzz脚本

use LWP 5.64;
$browser = LWP::UserAgent->new;
for($i=0;$i<256;$i++)
{
    $j=data10to16($i);
    print "[+] $i $j ".hex($j)." ".chr($i)."\n"
    # do GET/POST fuzz
}

sub data10to16{
    my($data)=@_;
    my $data16=sprintf("%x",$data);
    return $data16;
}
